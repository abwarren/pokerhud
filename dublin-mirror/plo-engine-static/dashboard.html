# Deploy to Cape Town (nuts4poker.com)

# Deploy to Dublin (potlimitomaha.xyz)
scp -i /home/ploxyz.pem -r ../static/* ubuntu@52.16.14.220:/opt/plo-equity/static/
ssh -i /home/ploxyz.pem ubuntu@52.16.14.220 "cp /opt/plo-equity/static/index.html /opt/plo-equity/static/engine-index.html"

# Deploy engine scripts to Dublin
scp -i /home/ploxyz.pem ~/2_DEVELOPMENT/scripts/plo/engine_scripts/*.py ubuntu@52.16.14.220:/opt/plo-engine-backend/scripts/

# Restart engine backend on Dublin
ssh -i /home/ploxyz.pem ubuntu@52.16.14.220 "kill $(pgrep -f 'plo-engine-backend.*gunicorn') && cd /opt/plo-engine-backend && nohup venv/bin/gunicorn -w 1 -k eventlet --bind 127.0.0.1:5050 --timeout 300 app:app &"

# Reload nginx on Dublin
ssh -i /home/ploxyz.pem ubuntu@52.16.14.220 "docker exec nginx-proxy nginx -s reload"
